function detectIsDarkMode(){
    const body = document.body;
    if(body?.classList?.contains("theme-dark")) return true;
    if(body?.classList?.contains("theme-light")) return false;
    const computedBg = window.getComputedStyle(document.documentElement).backgroundColor || "";
    const ch = computedBg.match(/\d+(?:\.\d+)?/g);
    if(ch && ch.length >= 3){
        const lum = (Number(ch[0])*299 + Number(ch[1])*587 + Number(ch[2])*114)/1000;
        return lum < 128;
    }
    return true;
}
