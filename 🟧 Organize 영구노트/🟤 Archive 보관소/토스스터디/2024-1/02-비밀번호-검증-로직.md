# 비밀번호 검증 로직

## 개요
세 가지 요구사항을 만족하는 비밀번호 검증 로직의 구현 방법입니다.

## 요구사항

1. ✅ 최근 5개 비밀번호 재사용 금지
2. ✅ 3개월 이내 사용 비밀번호 재사용 금지
3. ✅ 100명 이상 사용 중인 비밀번호 사용 금지

## 핵심 서비스 구현

### PasswordResetService

```java
@Service
@Slf4j
@Transactional
public class PasswordResetService {
    
    private final PasswordHistoryRepository historyRepository;
    private final PasswordUsageCountRepository usageCountRepository;
    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    private final CommonPasswordBloomFilter bloomFilter;
    private final RedisTemplate<String, Integer> redisTemplate;
    
    private static final int MAX_PASSWORD_USAGE = 100;
    private static final int RECENT_PASSWORD_LIMIT = 5;
    private static final int MONTH_LIMIT = 3;
    
    @Autowired
    public PasswordResetService(
            PasswordHistoryRepository historyRepository,
            PasswordUsageCountRepository usageCountRepository,
            UserRepository userRepository,
            PasswordEncoder passwordEncoder,
            CommonPasswordBloomFilter bloomFilter,
            RedisTemplate<String, Integer> redisTemplate) {
        this.historyRepository = historyRepository;
        this.usageCountRepository = usageCountRepository;
        this.userRepository = userRepository;
        this.passwordEncoder = passwordEncoder;
        this.bloomFilter = bloomFilter;
        this.redisTemplate = redisTemplate;
    }
    
    /**
     * 비밀번호 초기화 메인 로직
     */
    public void resetPassword(Long userId, String newPassword) {
        log.info("Password reset requested for user: {}", userId);
        
        String newPasswordHash = passwordEncoder.encode(newPassword);
        
        // 1단계: 최근 5개 비밀번호 검증 (가장 빠름)
        validateRecentPasswords(userId, newPassword);
        
        // 2단계: 3개월 이내 사용 비밀번호 검증
        validateThreeMonthHistory(userId, newPassword);
        
        // 3단계: 과다 사용 비밀번호 검증 (가장 복잡)
        validatePasswordUsageCount(newPasswordHash);
        
        // 4단계: 비밀번호 업데이트
        updatePassword(userId, newPassword, newPasswordHash);
        
        log.info("Password reset successful for user: {}", userId);
    }
    
    /**
     * 검증 1: 최근 5개 비밀번호 재사용 금지
     */
    private void validateRecentPasswords(Long userId, String newPassword) {
        List<PasswordHistory> recentPasswords = 
            historyRepository.findTop5ByUserIdOrderByUsedAtDesc(userId);
        
        for (PasswordHistory history : recentPasswords) {
            if (passwordEncoder.matches(newPassword, history.getPasswordHash())) {
                throw new InvalidPasswordException(
                    "최근 5개 비밀번호는 재사용할 수 없습니다.");
            }
        }
        
        log.debug("Recent password validation passed for user: {}", userId);
    }
    
    /**
     * 검증 2: 3개월 이내 사용 비밀번호 재사용 금지
     */
    private void validateThreeMonthHistory(Long userId, String newPassword) {
        LocalDateTime threeMonthsAgo = LocalDateTime.now().minusMonths(MONTH_LIMIT);
        
        List<PasswordHistory> threeMonthHistory = 
            historyRepository.findByUserIdAndUsedAtAfter(userId, threeMonthsAgo);
        
        for (PasswordHistory history : threeMonthHistory) {
            if (passwordEncoder.matches(newPassword, history.getPasswordHash())) {
                throw new InvalidPasswordException(
                    "3개월 이내 사용한 비밀번호는 재사용할 수 없습니다.");
            }
        }
        
        log.debug("Three-month history validation passed for user: {}", userId);
    }
    
    /**
     * 검증 3: 100명 이상 사용 중인 비밀번호 사용 금지
     * 
     * 3단계 검증 전략:
     * 1) Bloom Filter - 빠른 음성 판정
     * 2) Redis Cache - 빠른 양성 판정
     * 3) DB 조회 - 최종 확인
     */
    private void validatePasswordUsageCount(String passwordHash) {
        // Step 1: Bloom Filter로 빠른 필터링
        if (!bloomFilter.mightBeCommonPassword(passwordHash)) {
            // 확실히 100명 미만 사용 → 즉시 통과
            log.debug("Bloom filter: password is not common");
            return;
        }
        
        log.debug("Bloom filter: password might be common, checking cache");
        
        // Step 2: Redis 캐시 확인
        String cacheKey = "pwd:count:" + passwordHash;
        Integer cachedCount = redisTemplate.opsForValue().get(cacheKey);
        
        if (cachedCount != null) {
            if (cachedCount >= MAX_PASSWORD_USAGE) {
                throw new InvalidPasswordException(
                    "너무 많은 사용자가 사용 중인 비밀번호입니다.");
            }
            log.debug("Cache hit: usage count = {}", cachedCount);
            return;
        }
        
        log.debug("Cache miss, querying database");
        
        // Step 3: DB 조회 (캐시 미스)
        PasswordUsageCount usageCount = 
            usageCountRepository.findById(passwordHash).orElse(null);
        
        if (usageCount != null) {
            int count = usageCount.getUsageCount();
            
            // 캐시 갱신 (TTL: 24시간)
            redisTemplate.opsForValue().set(
                cacheKey, 
                count,
                Duration.ofHours(24)
            );
            
            if (count >= MAX_PASSWORD_USAGE) {
                throw new InvalidPasswordException(
                    "너무 많은 사용자가 사용 중인 비밀번호입니다.");
            }
            
            log.debug("DB check: usage count = {}", count);
        }
    }
    
    /**
     * 비밀번호 업데이트 및 이력 관리
     */
    private void updatePassword(Long userId, String newPassword, String newPasswordHash) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));
        
        String oldPasswordHash = user.getCurrentPassword();
        
        // 1. 이전 비밀번호의 사용 카운트 감소
        decrementPasswordUsage(oldPasswordHash);
        
        // 2. 사용자의 비밀번호 업데이트
        user.setCurrentPassword(newPasswordHash);
        userRepository.save(user);
        
        // 3. 비밀번호 이력에 추가
        PasswordHistory history = PasswordHistory.builder()
            .userId(userId)
            .passwordHash(newPasswordHash)
            .usedAt(LocalDateTime.now())
            .build();
        historyRepository.save(history);
        
        // 4. 새 비밀번호의 사용 카운트 증가
        incrementPasswordUsage(newPasswordHash);
    }
    
    /**
     * 비밀번호 사용 카운트 증가
     */
    private void incrementPasswordUsage(String passwordHash) {
        usageCountRepository.incrementOrCreate(passwordHash);
        
        // 캐시 무효화
        redisTemplate.delete("pwd:count:" + passwordHash);
        
        log.debug("Incremented usage count for password hash");
    }
    
    /**
     * 비밀번호 사용 카운트 감소
     */
    private void decrementPasswordUsage(String passwordHash) {
        usageCountRepository.decrement(passwordHash);
        
        // 캐시 무효화
        redisTemplate.delete("pwd:count:" + passwordHash);
        
        log.debug("Decremented usage count for password hash");
    }
}
```

## Repository 인터페이스

### PasswordHistoryRepository

```java
@Repository
public interface PasswordHistoryRepository extends JpaRepository<PasswordHistory, Long> {
    
    /**
     * 최근 5개 비밀번호 조회
     */
    @Query("SELECT ph FROM PasswordHistory ph " +
           "WHERE ph.userId = :userId " +
           "ORDER BY ph.usedAt DESC")
    List<PasswordHistory> findTop5ByUserIdOrderByUsedAtDesc(
        @Param("userId") Long userId, 
        Pageable pageable
    );
    
    default List<PasswordHistory> findTop5ByUserIdOrderByUsedAtDesc(Long userId) {
        return findTop5ByUserIdOrderByUsedAtDesc(userId, PageRequest.of(0, 5));
    }
    
    /**
     * 특정 기간 이후 비밀번호 이력 조회
     */
    @Query("SELECT ph FROM PasswordHistory ph " +
           "WHERE ph.userId = :userId " +
           "AND ph.usedAt >= :since " +
           "ORDER BY ph.usedAt DESC")
    List<PasswordHistory> findByUserIdAndUsedAtAfter(
        @Param("userId") Long userId,
        @Param("since") LocalDateTime since
    );
}
```

### PasswordUsageCountRepository

```java
@Repository
public interface PasswordUsageCountRepository extends JpaRepository<PasswordUsageCount, String> {
    
    /**
     * 비밀번호 사용 카운트 증가 또는 생성
     */
    @Modifying
    @Query(value = "INSERT INTO password_usage_count (password_hash, usage_count) " +
                   "VALUES (:passwordHash, 1) " +
                   "ON DUPLICATE KEY UPDATE usage_count = usage_count + 1",
           nativeQuery = true)
    void incrementOrCreate(@Param("passwordHash") String passwordHash);
    
    /**
     * 비밀번호 사용 카운트 감소
     */
    @Modifying
    @Query(value = "UPDATE password_usage_count " +
                   "SET usage_count = GREATEST(usage_count - 1, 0) " +
                   "WHERE password_hash = :passwordHash",
           nativeQuery = true)
    void decrement(@Param("passwordHash") String passwordHash);
    
    /**
     * 특정 사용 횟수 이상인 비밀번호 조회
     */
    @Query("SELECT puc.passwordHash FROM PasswordUsageCount puc " +
           "WHERE puc.usageCount >= :minCount")
    List<String> findPasswordHashesWithUsageGreaterThan(
        @Param("minCount") int minCount
    );
}
```

```java
        super(message);
    }

    public UserNotFoundException(String message) {
        super(message);
    }
}
```

### 글로벌 예외 핸들러

```java
@RestControllerAdvice
public class PasswordResetExceptionHandler {
    
    @ExceptionHandler(InvalidPasswordException.class)
    public ResponseEntity<ErrorResponse> handleInvalidPassword(
            InvalidPasswordException e) {
        
        ErrorResponse error = ErrorResponse.builder()
            .code("INVALID_PASSWORD")
            .message(e.getMessage())
            .timestamp(LocalDateTime.now())
            .build();
        
        return ResponseEntity
            .status(HttpStatus.BAD_REQUEST)
            .body(error);
    }
    
    @ExceptionHandler(UserNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleUserNotFound(
            UserNotFoundException e) {
        
        ErrorResponse error = ErrorResponse.builder()
            .code("USER_NOT_FOUND")
            .message(e.getMessage())
            .timestamp(LocalDateTime.now())
            .build();
        
        return ResponseEntity
            .status(HttpStatus.NOT_FOUND)
            .body(error);
    }
}
```

## 성능 최적화 포인트

### 1. 검증 순서 최적화
```
가장 빠른 검증부터 수행:
1) 최근 5개 검증 (최대 5번 해시 비교)
2) 3개월 이력 검증 (파티션 프루닝 활용)
3) 사용 횟수 검증 (Bloom Filter → Cache → DB)
```

### 2. 인덱스 활용
```sql
-- 최근 5개: idx_user_recent (user_id, used_at DESC)
EXPLAIN SELECT * FROM password_history 
WHERE user_id = 123 ORDER BY used_at DESC LIMIT 5;

-- 3개월 이력: 파티션 프루닝 + idx_user_recent
EXPLAIN SELECT * FROM password_history 
WHERE user_id = 123 AND used_at >= '2024-07-19';
```

### 3. 캐시 히트율 모니터링
```java
@Aspect
@Component
public class CacheMonitoringAspect {
    
    @Around("execution(* validatePasswordUsageCount(..))")
    public Object monitorCache(ProceedingJoinPoint joinPoint) throws Throwable {
        // 캐시 히트/미스 메트릭 수집
        return joinPoint.proceed();
    }
}
```

### 단위 테스트

```java
@SpringBootTest
class PasswordResetServiceTest {
    
    @Autowired
    private PasswordResetService passwordResetService;
    
    @Test
    @DisplayName("최근 5개 비밀번호 재사용 시 예외 발생")
    void testRecentPasswordRejection() {
        // Given
        Long userId = 1L;
        String recentPassword = "OldPassword123!";
        
        // When & Then
        assertThrows(InvalidPasswordException.class, () -> {
            passwordResetService.resetPassword(userId, recentPassword);
        });
    }
    
    @Test
    @DisplayName("3개월 이내 사용 비밀번호 재사용 시 예외 발생")
    void testThreeMonthHistoryRejection() {
        // Given
        Long userId = 1L;
        String recentPassword = "TwoMonthsAgoPassword!";
        
        // When & Then
        assertThrows(InvalidPasswordException.class, () -> {
            passwordResetService.resetPassword(userId, recentPassword);
        });
    }
    
    @Test
    @DisplayName("100명 이상 사용 중인 비밀번호 거부")
    void testCommonPasswordRejection() {
        // Given
        String commonPassword = "Password123!";
        
        // When & Then
        assertThrows(InvalidPasswordException.class, () -> {
            passwordResetService.resetPassword(1L, commonPassword);
        });
    }
    
    @Test
    @DisplayName("정상적인 비밀번호 변경 성공")
    void testSuccessfulPasswordReset() {
        // Given
        Long userId = 1L;
        String newPassword = "NewSecurePassword123!";
        
        // When & Then
        assertDoesNotThrow(() -> {
            passwordResetService.resetPassword(userId, newPassword);
        });
    }
}
```
