## 1. 비밀번호 사용 카운트 동기화

### 전체 재구축 배치

```java
@Component
@Slf4j
public class PasswordUsageCountRebuildBatch {
    
    private final UserRepository userRepository;
    private final PasswordUsageCountRepository usageCountRepository;
    private final JdbcTemplate jdbcTemplate;
    
    private static final int BATCH_SIZE = 10_000;
    private static final int TOTAL_USERS = 100_000_000; // 1억
    
    /**
     * 매일 새벽 2시 전체 재구축
     */
    @Scheduled(cron = "0 0 2 * * *")
    @Transactional
    public void rebuildPasswordUsageCount() {
        log.info("Starting password usage count rebuild");
        
        long startTime = System.currentTimeMillis();
        
        // 1. 기존 데이터 백업 (선택사항)
        backupCurrentData();
        
        // 2. 임시 테이블 생성
        createTempTable();
        
        // 3. 청크 단위로 처리
        for (long offset = 0; offset < TOTAL_USERS; offset += BATCH_SIZE) {
            processChunk(offset, BATCH_SIZE);
            
            if (offset % 100_000 == 0) {
                log.info("Progress: {}/{} ({:.1f}%)", 
                         offset, TOTAL_USERS, 
                         (double) offset / TOTAL_USERS * 100);
            }
        }
        
        // 4. 원자적 테이블 교체
        swapTables();
        
        long duration = System.currentTimeMillis() - startTime;
        log.info("Password usage count rebuild completed in {}ms", duration);
    }
    
    private void processChunk(long offset, int limit) {
        // 청크 단위로 사용자 조회
        List<User> users = userRepository.findAllWithOffset(offset, limit);
        
        // 비밀번호별 카운트 집계
        Map<String, Integer> countMap = new HashMap<>();
        for (User user : users) {
            countMap.merge(user.getCurrentPassword(), 1, Integer::sum);
        }
        
        // 임시 테이블에 일괄 삽입/업데이트
        batchUpsertToTempTable(countMap);
    }
    
    @Transactional
    protected void batchUpsertToTempTable(Map<String, Integer> countMap) {
        jdbcTemplate.batchUpdate(
            "INSERT INTO password_usage_count_temp (password_hash, usage_count) " +
            "VALUES (?, ?) " +
            "ON DUPLICATE KEY UPDATE usage_count = usage_count + VALUES(usage_count)",
            new BatchPreparedStatementSetter() {
                private final List<Map.Entry<String, Integer>> entries = 
                    new ArrayList<>(countMap.entrySet());
                
                @Override
                public void setValues(PreparedStatement ps, int i) throws SQLException {
                    Map.Entry<String, Integer> entry = entries.get(i);
                    ps.setString(1, entry.getKey());
                    ps.setInt(2, entry.getValue());
                }
                
                @Override
                public int getBatchSize() {
                    return entries.size();
                }
            }
        );
    }
    
    private void createTempTable() {
        jdbcTemplate.execute(
            "CREATE TABLE IF NOT EXISTS password_usage_count_temp " +
            "LIKE password_usage_count"
        );
        jdbcTemplate.execute("TRUNCATE TABLE password_usage_count_temp");
    }
    
    private void swapTables() {
        jdbcTemplate.execute(
            "RENAME TABLE " +
            "password_usage_count TO password_usage_count_old, " +
            "password_usage_count_temp TO password_usage_count, " +
            "password_usage_count_old TO password_usage_count_temp"
        );
        jdbcTemplate.execute("DROP TABLE password_usage_count_temp");
    }
    
    private void backupCurrentData() {
        String timestamp = LocalDateTime.now().format(
            DateTimeFormatter.ofPattern("yyyyMMdd_HHmmss"));
        
        jdbcTemplate.execute(
            "CREATE TABLE password_usage_count_backup_" + timestamp + 
            " AS SELECT * FROM password_usage_count"
        );
    }
}
```

## 2. 증분 업데이트 전략

### 실시간 이벤트 기반 업데이트

```java
@Component
@Slf4j
public class PasswordUsageIncrementalUpdater {
    
    private final PasswordUsageCountRepository repository;
    private final CommonPasswordBloomFilter bloomFilter;
    private final RedisTemplate<String, Integer> redisTemplate;
    
    /**
     * 비밀번호 변경 이벤트 처리
     */
    @TransactionalEventListener(phase = TransactionPhase.AFTER_COMMIT)
    @Async
    public void onPasswordChanged(PasswordChangedEvent event) {
        String oldHash = event.getOldPasswordHash();
        String newHash = event.getNewPasswordHash();
        
        // 1. DB 카운트 업데이트
        decrementUsageCount(oldHash);
        incrementUsageCount(newHash);
        
        // 2. 캐시 무효화
        invalidateCache(oldHash);
        invalidateCache(newHash);
        
        // 3. Bloom Filter 업데이트 (필요시)
        updateBloomFilterIfNeeded(newHash);
    }
    
    @Retryable(maxAttempts = 3, backoff = @Backoff(delay = 1000))
    private void incrementUsageCount(String passwordHash) {
        int newCount = repository.incrementOrCreate(passwordHash);
        
        log.debug("Incremented usage count for password: new count = {}", newCount);
    }
    
    @Retryable(maxAttempts = 3, backoff = @Backoff(delay = 1000))
    private void decrementUsageCount(String passwordHash) {
        repository.decrement(passwordHash);
        
        log.debug("Decremented usage count for password");
    }
    
    private void invalidateCache(String passwordHash) {
        String cacheKey = "pwd:count:" + passwordHash;
        redisTemplate.delete(cacheKey);
    }
    
    private void updateBloomFilterIfNeeded(String passwordHash) {
        // 100명 도달 시 Bloom Filter 추가
        Integer count = repository.findById(passwordHash)
            .map(PasswordUsageCount::getUsageCount)
            .orElse(0);
        
        if (count >= 100) {
            bloomFilter.addCommonPassword(passwordHash);
        }
    }
}
```

```java
@Component
public class PasswordHistoryArchiver {
    
    private final PasswordHistoryRepository historyRepository;
    private final JdbcTemplate jdbcTemplate;
    
    /**
     * 매월 1일 1년 이상 오래된 이력 아카이빙
     */
    @Scheduled(cron = "0 0 3 1 * *")
    @Transactional
    public void archiveOldHistory() {
        log.info("Starting password history archiving");
        
        LocalDateTime oneYearAgo = LocalDateTime.now().minusYears(1);
        
        // 1. 아카이브 테이블로 이동
        int archived = jdbcTemplate.update(
            "INSERT INTO password_history_archive " +
            "SELECT * FROM password_history " +
            "WHERE used_at < ?",
            oneYearAgo
        );
        
        // 2. 원본 테이블에서 삭제
        int deleted = jdbcTemplate.update(
            "DELETE FROM password_history WHERE used_at < ?",
            oneYearAgo
        );
        
        log.info("Archived {} records, deleted {} records", archived, deleted);
    }
    
    /**
     * 아카이브 테이블 압축
     */
    @Scheduled(cron = "0 0 4 1 * *")
    public void compressArchive() {
        jdbcTemplate.execute(
            "ALTER TABLE password_history_archive ENGINE=ARCHIVE"
        );
        
        log.info("Archive table compressed");
    }
}
```

```java
@Component
public class PartitionManager {
    
    private final JdbcTemplate jdbcTemplate;
    
    /**
     * 매월 1일 새 파티션 추가
     */
    @Scheduled(cron = "0 0 1 1 * *")
    public void addMonthlyPartition() {
        LocalDateTime nextMonth = LocalDateTime.now().plusMonths(1);
        String partitionName = "p" + nextMonth.format(DateTimeFormatter.ofPattern("yyyyMM"));
        int partitionValue = Integer.parseInt(
            nextMonth.format(DateTimeFormatter.ofPattern("yyyyMM"))
        );
        
        String sql = String.format(
            "ALTER TABLE password_history ADD PARTITION " +
            "(PARTITION %s VALUES LESS THAN (%d))",
            partitionName, partitionValue + 1
        );
        
        jdbcTemplate.execute(sql);
        
        log.info("Added new partition: {}", partitionName);
    }
    
    /**
     * 오래된 파티션 삭제
     */
    @Scheduled(cron = "0 0 2 1 * *")
    public void dropOldPartitions() {
        LocalDateTime oneYearAgo = LocalDateTime.now().minusYears(1);
        String oldPartitionName = "p" + oneYearAgo.format(
            DateTimeFormatter.ofPattern("yyyyMM")
        );
        
        String sql = String.format(
            "ALTER TABLE password_history DROP PARTITION %s",
            oldPartitionName
        );
        
        try {
            jdbcTemplate.execute(sql);
            log.info("Dropped old partition: {}", oldPartitionName);
        } catch (Exception e) {
            log.warn("Failed to drop partition {}: {}", oldPartitionName, e.getMessage());
        }
    }
}
```

## 4. 데이터 정합성 검증

### 주기적 정합성 체크

```java
@Component
public class DataConsistencyChecker {
    
    private final UserRepository userRepository;
    private final PasswordUsageCountRepository usageCountRepository;
    private final JdbcTemplate jdbcTemplate;
    
    /**
     * 매주 일요일 새벽 정합성 검증
     */
    @Scheduled(cron = "0 0 5 * * SUN")
    public void checkConsistency() {
        log.info("Starting data consistency check");
        
        // 1. 사용자 테이블과 카운트 테이블 비교
        List<InconsistentPassword> inconsistencies = findInconsistencies();
        
        if (!inconsistencies.isEmpty()) {
            log.warn("Found {} inconsistencies", inconsistencies.size());
            
            // 2. 불일치 데이터 수정
            fixInconsistencies(inconsistencies);
        } else {
            log.info("No inconsistencies found");
        }
    }
    
    private List<InconsistentPassword> findInconsistencies() {
        String sql = 
            "SELECT " +
            "    u.current_password as password_hash, " +
            "    COUNT(*) as actual_count, " +
            "    COALESCE(puc.usage_count, 0) as recorded_count " +
            "FROM users u " +
            "LEFT JOIN password_usage_count puc " +
            "    ON u.current_password = puc.password_hash " +
            "GROUP BY u.current_password " +
            "HAVING actual_count != recorded_count";
        
        return jdbcTemplate.query(sql, (rs, rowNum) -> 
            new InconsistentPassword(
                rs.getString("password_hash"),
                rs.getInt("actual_count"),
                rs.getInt("recorded_count")
            )
        );
    }
    
    @Transactional
    protected void fixInconsistencies(List<InconsistentPassword> inconsistencies) {
        for (InconsistentPassword inc : inconsistencies) {
            jdbcTemplate.update(
                "INSERT INTO password_usage_count (password_hash, usage_count) " +
                "VALUES (?, ?) " +
                "ON DUPLICATE KEY UPDATE usage_count = ?",
                inc.getPasswordHash(),
                inc.getActualCount(),
                inc.getActualCount()
            );
        }
        
        log.info("Fixed {} inconsistencies", inconsistencies.size());
    }
}
```

## 5. Spring Batch 구현

### 대용량 배치 작업

```java
@Configuration
@EnableBatchProcessing
public class PasswordBatchConfig {
    
    @Bean
    public Job passwordUsageCountJob(JobBuilderFactory jobBuilderFactory,
                                      Step rebuildStep) {
        return jobBuilderFactory.get("passwordUsageCountJob")
            .incrementer(new RunIdIncrementer())
            .start(rebuildStep)
            .build();
    }
    
    @Bean
    public Step rebuildStep(StepBuilderFactory stepBuilderFactory,
                            ItemReader<User> reader,
                            ItemProcessor<User, PasswordCount> processor,
                            ItemWriter<PasswordCount> writer) {
        return stepBuilderFactory.get("rebuildStep")
            .<User, PasswordCount>chunk(10_000)
            .reader(reader)
            .processor(processor)
            .writer(writer)
            .build();
    }
    
    @Bean
    @StepScope
    public JdbcPagingItemReader<User> userReader(DataSource dataSource) {
        JdbcPagingItemReader<User> reader = new JdbcPagingItemReader<>();
        reader.setDataSource(dataSource);
        reader.setPageSize(10_000);
        
        MySqlPagingQueryProvider queryProvider = new MySqlPagingQueryProvider();
        queryProvider.setSelectClause("SELECT user_id, current_password");
        queryProvider.setFromClause("FROM users");
        queryProvider.setSortKeys(Map.of("user_id", Order.ASCENDING));
        
        reader.setQueryProvider(queryProvider);
        reader.setRowMapper((rs, rowNum) -> {
            User user = new User();
            user.setUserId(rs.getLong("user_id"));
            user.setCurrentPassword(rs.getString("current_password"));
            return user;
        });
        
        return reader;
    }
    
    @Bean
    public ItemProcessor<User, PasswordCount> passwordCountProcessor() {
        return user -> {
            PasswordCount count = new PasswordCount();
            count.setPasswordHash(user.getCurrentPassword());
            count.setCount(1);
            return count;
        };
    }
    
    @Bean
    public ItemWriter<PasswordCount> passwordCountWriter(JdbcTemplate jdbcTemplate) {
        return items -> {
            // 비밀번호별로 그룹핑
            Map<String, Integer> countMap = items.stream()
                .collect(Collectors.groupingBy(
                    PasswordCount::getPasswordHash,
                    Collectors.summingInt(PasswordCount::getCount)
                ));
            
            // 배치 업데이트
            jdbcTemplate.batchUpdate(
                "INSERT INTO password_usage_count_temp (password_hash, usage_count) " +
                "VALUES (?, ?) " +
                "ON DUPLICATE KEY UPDATE usage_count = usage_count + VALUES(usage_count)",
                new BatchPreparedStatementSetter() {
                    private final List<Map.Entry<String, Integer>> entries = 
                        new ArrayList<>(countMap.entrySet());
                    
                    @Override
                    public void setValues(PreparedStatement ps, int i) throws SQLException {
                        ps.setString(1, entries.get(i).getKey());
                        ps.setInt(2, entries.get(i).getValue());
                    }
                    
                    @Override
                    public int getBatchSize() {
                        return entries.size();
                    }
                }
            );
        };
    }
}
```

## 6. 동시성 제어

### 분산 락을 이용한 동시성 제어

```java
@Component
public class DistributedLockManager {
    
    private final RedissonClient redissonClient;
    
    /**
     * 분산 락을 이용한 안전한 카운트 업데이트
     */
    public void updateWithLock(String passwordHash, Runnable updateTask) {
        String lockKey = "lock:pwd:" + passwordHash;
        RLock lock = redissonClient.getLock(lockKey);
        
        try {
            // 락 획득 (최대 10초 대기, 30초 후 자동 해제)
            if (lock.tryLock(10, 30, TimeUnit.SECONDS)) {
                try {
                    updateTask.run();
                } finally {
                    lock.unlock();
                }
            } else {
                throw new LockAcquisitionException(
                    "Failed to acquire lock for: " + passwordHash);
            }
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new RuntimeException("Interrupted while acquiring lock", e);
        }
    }
}
```

```java
@Component
public class BatchJobMonitor {
    
    private final JobExplorer jobExplorer;
    private final MeterRegistry meterRegistry;
    
    /**
     * 배치 작업 상태 모니터링
     */
    @Scheduled(fixedRate = 60000) // 1분마다
    public void monitorBatchJobs() {
        List<String> jobNames = jobExplorer.getJobNames();
        
        for (String jobName : jobNames) {
            List<JobInstance> instances = jobExplorer.getJobInstances(jobName, 0, 10);
            
            for (JobInstance instance : instances) {
                List<JobExecution> executions = jobExplorer.getJobExecutions(instance);
                
                for (JobExecution execution : executions) {
                    recordMetrics(jobName, execution);
                    
                    if (execution.getStatus() == BatchStatus.FAILED) {
                        alertOnFailure(jobName, execution);
                    }
                }
            }
        }
    }
    
    private void recordMetrics(String jobName, JobExecution execution) {
        meterRegistry.gauge("batch.job.duration", 
            Tags.of("job", jobName),
            execution.getEndTime() != null ? 
                Duration.between(execution.getStartTime(), execution.getEndTime()).toMillis() : 0
        );
    }
    
    private void alertOnFailure(String jobName, JobExecution execution) {
        log.error("Batch job failed: {} - {}", 
                  jobName, 
                  execution.getAllFailureExceptions());
        
        // 슬랙, 이메일 등으로 알림 전송
        sendAlert(jobName, execution);
    }

        // 알림 로직 구현
    }
}
```

### 병렬 처리

```java
@Configuration
public class ParallelBatchConfig {
    
    @Bean
    public TaskExecutor batchTaskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(10);
        executor.setMaxPoolSize(20);
        executor.setQueueCapacity(50);
        executor.setThreadNamePrefix("batch-");
        executor.initialize();
        return executor;
    }
    
    @Bean
    public Step parallelRebuildStep(StepBuilderFactory stepBuilderFactory,
                                     PartitionHandler partitionHandler,
                                     Step workerStep) {
        return stepBuilderFactory.get("parallelRebuildStep")
            .partitioner("workerStep", partitioner())
            .partitionHandler(partitionHandler)
            .build();
    }
    
    @Bean
    public Partitioner partitioner() {
        return gridSize -> {
            Map<String, ExecutionContext> partitions = new HashMap<>();
            
            int chunkSize = 100_000_000 / gridSize; // 1억 / 파티션 수
            
            for (int i = 0; i < gridSize; i++) {
                ExecutionContext context = new ExecutionContext();
                context.putLong("minValue", i * chunkSize);
                context.putLong("maxValue", (i + 1) * chunkSize);
                partitions.put("partition" + i, context);
            }
            
            return partitions;
        };
    }
}
```
