### 리스트

```python
리스트명 = [요소1, 요소2, 요소3, ...]

a = [1, 2, ['Life', 'is']]
b = list() # 빈 리스트

# 슬라이싱
>>> a[-1][0]
'Life'

# 더하기
>>> a = [1, 2, 3]
>>> b = [4, 5, 6]
>>> a + b
[1, 2, 3, 4, 5, 6]

# 곱하기
>>> a = [1, 2, 3]
>>> a * 3
[1, 2, 3, 1, 2, 3, 1, 2, 3]

# 길이
>>> len(a)
3

# 수정
>>> a = [1, 2, 3]
>>> a[2] = 4
>>> a
[1, 2, 4]

# 삭제 O(n)
>>> a = [1, 2, 3]
>>> del a[1]
>>> a
[1, 3]

## 문자열 -> 리스트
s = input()
l = list(s)
print(l)
```

### 리스트 함수

```python
# append O(1)
>>> a = [1, 2, 3]
>>> a.append(4) # 메모리 재할당이 일어나기 때문에 조심!
>>> a
[1, 2, 3, 4]
>>> a.append([5, 6])
>>> a
[1, 2, 3, 4, [5, 6]]

# sort O(NlogN)
>>> a = [1, 4, 3, 2]
>>> a.sort()
>>> a
[1, 2, 3, 4]
>>> a.sort(reverse = True) # 내림차순 정렬
>>> a.sort(key=len) # 정렬을 목적으로 하는 함수를 값으로 넣는다.
>>> # (len은 문자열일때 가능)

# reverse
>>> a = ['a', 'c', 'b']
>>> a.reverse()
>>> a
['b', 'c', 'a']

# index # O(n)
>>> a = [1, 2, 3]
>>> a.index(3) 
2
>>> a.index(1)
0
>>> a.index(0) # 오류 발생

# insert
>>> a = [1, 2, 3]
>>> a.insert(0, 4)
>>> a
[4, 1, 2, 3]

# remove
>>> a = [1, 2, 3, 1, 2, 3]
>>> a.remove(3) # 첫번째 3만 제거
>>> a
[1, 2, 1, 2, 3]

# pop O(n) -> deque로 해결하기!
>>> a = [1, 2, 3]
>>> a.pop() # 마지막 요소 리턴, 삭제
3
>>> a
[1, 2]

>>> a = [1, 2, 3]
>>> a.pop(1)
2
>>> a
[1, 3]

# count O(n)
>>> a = [1, 2, 3, 1]
>>> a.count(1)
2

# extend
>>> a = [1, 2, 3]
>>> a.extend([4, 5]) # 리스트만 올 수 있다 a += [4,5]와 같음
>>> a
[1, 2, 3, 4, 5]
```
