```sql
ADD FULLTEXT INDEX ft_idx_title_content (title, content);
```

#### 고급 검색 쿼리
```java
@Service
public class RoomSearchService {
    @Autowired
    private ElasticsearchOperations elasticsearchOperations;
    
    public SearchHits<RoomDocument> search(SearchRequest request) {
        NativeSearchQuery query = NativeSearchQuery.builder()
            .withQuery(q -> q
                .bool(b -> b
                    .must(m -> m
                        .multiMatch(mm -> mm
                            .query(request.getKeyword())
                            .fields("title^3", "content^1")  // title 가중치 3배
                            .fuzziness("AUTO")
                        )
                    )
                    .filter(f -> f
                        .term(t -> t
                            .field("category")
                            .value(request.getCategory())
                        )
                    )
                )
            )
            .withSort(s -> s
                .score(sc -> sc.order(SortOrder.Desc))
            )
            .withPageable(PageRequest.of(0, 20))
            .build();
            
        return elasticsearchOperations.search(query, RoomDocument.class);
    }
}
```

#### 실시간 동기화 (Change Data Capture)
```java
// Kafka를 통한 실시간 동기화
@KafkaListener(topics = "room-events")
public void syncToElasticsearch(RoomEvent event) {
    switch (event.getType()) {
        case CREATED:
            indexRoom(event.getRoom());
            break;
        case UPDATED:
            updateRoom(event.getRoom());
            break;
        case DELETED:
            deleteRoom(event.getRoomId());
            break;
    }
}

private void indexRoom(Room room) {
    RoomDocument doc = RoomDocument.fromEntity(room);
    roomSearchRepository.save(doc);
}
```

```java
public List<String> autocomplete(String prefix) {
    NativeSearchQuery query = NativeSearchQuery.builder()
        .withQuery(q -> q
            .match(m -> m
                .field("title.autocomplete")
                .query(prefix)
            )
        )
        .withSourceFilter(new FetchSourceFilter(new String[]{"title"}, null))
        .withPageable(PageRequest.of(0, 10))
        .build();
        
    return elasticsearchOperations.search(query, RoomDocument.class)
        .stream()
        .map(hit -> hit.getContent().getTitle())
        .distinct()
        .collect(Collectors.toList());
}
```
